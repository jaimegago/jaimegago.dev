{{- $src := .Get "src" -}} {{- $alt := .Get "alt" -}} {{- $caption := .Get "caption" -}} {{- $thumb := eq (.Get "thumb") "true" -}} {{- $margin := eq (.Get "margin") "true" -}} {{- $breakout := eq (.Get "breakout") "true" -}} {{- $width := .Get "width" | default "380x" -}} {{- $cap := .Get "cap" -}} {{- /* float: an in-column figure the text wraps around. Accepts a side ("left"/"right") or "true" (defaults right). Any other value is off. */ -}} {{- $floatRaw := .Get "float" -}} {{- $floatSide := "" -}} {{- if or (eq $floatRaw "right") (eq $floatRaw "true") -}} {{- $floatSide = "right" -}} {{- else if eq $floatRaw "left" -}} {{- $floatSide = "left" -}} {{- end -}} {{- $float := ne $floatSide "" -}} {{- /* optional float width as a percent of the measure (default 40 in CSS) */ -}} {{- $floatw := .Get "floatwidth" -}} {{- if not $src -}} {{- errorf "figure shortcode: missing required `src` (used at %s)" .Position -}} {{- end -}} {{- if or (not $alt) (eq (trim $alt " ") "") -}} {{- errorf "figure shortcode: `alt` is required and must be non-empty (used at %s)" .Position -}} {{- end -}} {{- if or (or $thumb $margin) $float -}} {{- $img := resources.Get $src -}} {{- if not $img -}} {{- errorf "figure shortcode: could not find asset `%s` (used at %s)" $src .Position -}} {{- end -}} {{- /* margin figures display at 176px in the rail lane; floated figures at ~40% of the 620px measure. Resize 2x for sharpness in both cases. */ -}} {{- $w := $width -}} {{- if $margin }}{{ $w = "400x" }}{{ else if $float }}{{ $w = "560x" }}{{ end -}} {{- $small := $img.Resize $w -}} {{- $cls := "figure-thumb" -}} {{- if $margin }}{{ $cls = "figure-margin" }}{{ else if $float }}{{ $cls = printf "figure-float figure-float--%s" $floatSide }}{{ end -}} {{- else -}} {{- end -}}
